def estPremier(taille)
        3.upto(taille) do |x|
                @nonpremier=0
                @tab.each do |object|
                        if (object.compareToi(x)==false)
                                @nonpremier=1
                        end
                end
                if(@nonpremier==0)
                        tab<<Filtre.initialiseFiltre(x)
                end
        end
end
